# 堆排序
```go
package main

import "fmt"

func main() {
	arr := []int{5, 2, 3, 4, 6, 9, 0, 1, 4}
    
	sort(arr)
    
	fmt.Println(arr)
}

func sort(arr []int) {
	heapify_build(arr)
	n := len(arr)
	for i := n-1; i > 0; i-- {
		arr[i], arr[0] = arr[0], arr[i]
		arr = arr[:i]
		heapify(arr, 0)
	}
}

func heapify_build(arr []int) {
	n := len(arr)
	for i := (n-1)/2; i >= 0; i-- {
		heapify(arr, i)
	}
}

func heapify(arr []int, k int) {
	n := len(arr)
	max := k
	l, r := 2*k+1, 2*k+2
	if l <= n-1 && arr[l] > arr[max] {
		max = l
	}
	if r <= n-1 && arr[r] > arr[max] {
		max = r
	}
	if max != k {
		arr[max], arr[k] = arr[k], arr[max]
		heapify(arr, max)
	}
}
```
