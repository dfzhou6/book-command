# 选择
```go
package main

import "fmt"

func main() {
	arr := []int{5, 2, 3, 4, 6, 9, 0, 1, 4}
    
	sort(arr)
    
	fmt.Println(arr)
}

func sort(arr []int) {
	n := len(arr)
	// 第一层循环：选择次数
	for i := 0; i < n-1; i++ {
		// 第二层循环：以当前元素为标的，从其余的元素中选最小(通过下标的形式)
		k := i
		for j := i+1; j < n; j++ {
			if arr[j] < arr[k] {
				k = j
			}
		}
		arr[i], arr[k] = arr[k], arr[i]
	}
}
```
